    <script>
        // Menú hamburguesa
        const burger = document.querySelector('.burger');
        const nav = document.querySelector('.nav-links');
        const navLinks = document.querySelectorAll('.nav-links li');

        burger.addEventListener('click', () => {
            nav.classList.toggle('nav-active');
            burger.classList.toggle('toggle');
        });

        // Cerrar menú al hacer clic en un enlace
        navLinks.forEach(link => {
            link.addEventListener('click', () => {
                nav.classList.remove('nav-active');
                burger.classList.remove('toggle');
            });
        });

        // Smooth scrolling para los enlaces
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();
                
                document.querySelector(this.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });

        // Efecto de carga
        document.addEventListener('DOMContentLoaded', () => {
            document.body.style.opacity = '1';
            
            // Crear muelitas que siguen el cursor
            const teethContainer = document.getElementById('teeth-container');
            const teethCount = 5;
            
            for (let i = 0; i < teethCount; i++) {
                const tooth = document.createElement('div');
                tooth.classList.add('tooth');
                tooth.style.width = `${30 - i * 3}px`;
                tooth.style.height = `${30 - i * 3}px`;
                tooth.style.opacity = `${0.7 - i * 0.1}`;
                teethContainer.appendChild(tooth);
            }
            
            const teeth = document.querySelectorAll('.tooth');
            let mouseX = 0;
            let mouseY = 0;
            let prevMouseX = 0;
            let prevMouseY = 0;
            
            // Capturar posición del mouse
            document.addEventListener('mousemove', (e) => {
                mouseX = e.clientX;
                mouseY = e.clientY;
            });
            
            // Animación de las muelitas
            function animateTeeth() {
                let posX = prevMouseX + (mouseX - prevMouseX) / 10;
                let posY = prevMouseY + (mouseY - prevMouseY) / 10;
                
                teeth.forEach((tooth, index) => {
                    const delay = index * 0.05;
                    
                    setTimeout(() => {
                        tooth.style.left = `${posX - tooth.offsetWidth / 2}px`;
                        tooth.style.top = `${posY - tooth.offsetHeight / 2}px`;
                        
                        // Actualizar posición para la siguiente muelita
                        posX += (mouseX - posX) / 5;
                        posY += (mouseY - posY) / 5;
                    }, delay * 1000);
                });
                
                prevMouseX = posX;
                prevMouseY = posY;
                
                requestAnimationFrame(animateTeeth);
            }
            
            animateTeeth();
        });
    </script>
